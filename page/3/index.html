校验签名是否被更改

Android apk在签名中的一对抗。

Java代码的方式

用Signature类中的hashCode()方法，校验签名是否被更改过。

try {
    PackageManager packageManager = this.getPackageManager();
    String packageName = this.getPackageName();
    PackageInfo packageInfo = packageManager.getPackageInfo(packageName, GET_SIGNATURES);
    Signature[] signatures = packageInfo.signatures;
    Signature signature = signatures[0];
    int hashCode = signature.hashCode();
    Log.v("NSSK_Log", "hash code : " + hashCode);
} catch (PackageManager.NameNotFoundException e) {
    e.printStackTrace();
}

在3种不同的签名的情况下，hashCode的结果是不一致的。

Java反射调用的方式

在Java代码的基础上，我们可以用反射调用的方式来提高难度。

#include <jni.h>
#include <string.h>

extern "C"
JNIEXPORT jint JNICALL
Java_org_nisosaikou_signaturejni_MainActivity_checkSignature(JNIEnv *env, jobject thiz)
{
    jclass cls_thiz = env->GetObjectClass(thiz);
    jmethodID mid_getPackageManager = env->GetMethodID(cls_thiz, "getPackageManager", "()Landroid/content/pm/PackageManager;");
    jobject obj_packageManager = env->CallObjectMethod(thiz, mid_getPackageManager);
    jmethodID mid_getPackageName = env->GetMethodID(cls_thiz, "getPackageName", "()Ljava/lang/String;");
    jstring packageName = (jstring)env->CallObjectMethod(thiz, mid_getPackageName);

    jclass cls_PackageManager = env->GetObjectClass(obj_packageManager);
    jmethodID mid_getPackageInfo = env->GetMethodID(cls_PackageManager, "getPackageInfo", "(Ljava/lang/String;I)Landroid/content/pm/PackageInfo;");
    jobject obj_packageInfo = env->CallObjectMethod(obj_packageManager, mid_getPackageInfo, packageName, 64);
    jclass cls_PackageInfo = env->GetObjectClass(obj_packageInfo);
    jfieldID fid_signatures = env->GetFieldID(cls_PackageInfo, "signatures", "[Landroid/content/pm/Signature;");
    jobjectArray objs_signatures =  (jobjectArray)env->GetObjectField(obj_packageInfo, fid_signatures);
    jobject obj_signature = env->GetObjectArrayElement(objs_signatures, 0);
    jclass cls_Signature = env->GetObjectClass(obj_signature);
    jmethodID mid_hashCode = env->GetMethodID(cls_Signature, "hashCode", "()I");
    jint hashCode = env->CallIntMethod(obj_signature, mid_hashCode);
    return hashCode;
}
